% Empty Fill Page if Prev. Chapter does not Reach Odd Page Before New Chapter
  % Redefine \cleardouble page, to remove header and footer (i.e. blank page).
  % I would have wanted to do this in the .sty file, but it is recommended not
  % to use \makeatletter in style or class files.
\makeatletter
\def\cleardoublepage{
  \clearpage
  \if@twoside
    \ifodd\c@page
    \else
      \hbox{}
      \thispagestyle{empty}
      \newpage
      \if@twocolumn
        \hbox{}
        \newpage
      \fi
    \fi
  \fi}
\makeatother
